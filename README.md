# Configure VIM Editor for Python
```
# vim ~/.vimrc

syntax enable
set tabstop=4
set shiftwidth=4
set expandtab
set number
filetype indent on
set autoindent
let python_highlight_all = 1

# mkdir -p ~/.vim/syntax
# wget https://raw.githubusercontent.com/hdima/python-syntax/master/syntax/python.vim -P ~/.vim/syntax
```
